def correct_line_breaks(text):
    lines = text.split('\n')
    corrected_lines = []
    combined_line = ""
    url_pattern = re.compile(r'^https?://')
    tag_start_pattern = re.compile(r'^<[^>]+$')
    tag_end_pattern = re.compile(r'.+>$')

    for line in lines:
        stripped_line = line.strip()
        
        # If the current line starts with a URL or a tag, combine it with the previous line
        if url_pattern.match(stripped_line) or tag_start_pattern.match(stripped_line):
            combined_line += stripped_line
        else:
            # If the previous combined line ends with a tag, just add the current line
            if tag_end_pattern.match(combined_line):
                combined_line += stripped_line
            else:
                # Otherwise, save the combined line and reset for the next block
                if combined_line:
                    corrected_lines.append(combined_line)
                combined_line = stripped_line

    # Append the last combined line if it exists
    if combined_line:
        corrected_lines.append(combined_line)

    return '\n'.join(corrected_lines)
